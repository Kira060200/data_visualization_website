Presupunem ca avem un experiment aleator. Ne interesam la realizarea unui eveniment A, considerand ca sansa de realizare a lui A este p. Aruncam o moneda cu probabilitatea de p (prob).

            prob = 0.33
            a = 0

            FctMasa = function(p){
                plot(c(1,0),c(p,1-p), lwd=5)
                segments(c(1,0), 0, c(1,0), c(p,1-p), col="red")
            }
            
     
            F = function(x,p){
                if (x<0){
                    y=0
                }else if (x<1){
                    y = 1-p
                }else{
                    y=1
                }
                return(y)
            }
            
            P = function(p, a, b=NULL, param=NULL)
            {
                if(is.null(b))
                {
                    if(is.null(param))
                    {
                        return(F(a, p))
                    }
                    else
                    {
                        return (1 - F(a, p))
                    }
                }
                else
                {
                    return (F(b, p) - F(a, p))
                }
            }
            
            fctMasa <- ({
                FctMasa(prob)
            })

            fctRep <- ({
                F2 = Vectorize(F, vectorize.args = "x")
                t2 = seq(-1,2,length.out = 1000)
                y2 = F2(t2,prob)
                plot(t2, y2, type= "l", col="red")
            })

            SelectProb = "P(x<=a)"
            
            
            fctProb <- ({
                
                F = Vectorize(F, vectorize.args = "x")
                t = seq(-1,2,length.out = 1000)
                y = F(t,prob)
                plot(t, y, type= "l", col="red")
                
                if(SelectProb=="P(x<=a)"){
                    x = seq(0,a)
                    y=F(x, prob)
              
                    
                    for(i in 1:length(x)){
                        tmp = 0
                        if(i==1)
                            tmp = 0
                        else
                            tmp = y[i-1]
                        lines(c(x[i],x[i]), c(tmp,y[i]), col="blue")
                    }
  
                }else if(SelectProb=="P(x>=b)"){
                    x = seq(b,2)
                    y = F(x,prob)
                    for(i in 1:length(x)){
                        tmp = 0
                        if(i==1)
                            tmp = 0
                        else
                            tmp = y[i-1]
                        lines(c(x[i],x[i]), c(tmp,y[i]), col="blue")
                    }
                }else{
                    x = seq(a,b)
                    y = F(x,prob)
                    for(i in 1:length(x)){
                        tmp = 0
                        if(i==1)
                            tmp = 0
                        else
                            tmp = y[i-1]
                        lines(c(x[i],x[i]), c(tmp,y[i]), col="blue")
                    }
                }
            
            })

            if(SelectProb=="P(x<=a)"){
                c("Probability: ", P(prob, a))
            }else if(SelectProb=="P(x>=b)"){
                c("Probability: ", P(prob, b, param = 1)+F(b, prob))

                c("Probability: ", P(prob, a, b)+F(a, prob))
